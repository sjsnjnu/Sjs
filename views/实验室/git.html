    git add
git rm filename #从暂存区和工作区删除
git rm --cached filename #只从暂存区移除
git status
git diff #暂存区与工作区
git diff --cached #暂存区与上一次提交
git diff branch1 branch2 #两个分支之间
git commit -m message #提交
git reset HEAD filename #撤销add
git reset HEAD^ #回退到上个版本
git reset --hard version  #git log可以查看version，只需要前7位哈希值
git branch sjs1
git checkout -b sjs1
git branch -v #查看所有分支
git branch -d sjs1 #删除
git branch -m old new #改名
git checkout sjs1 #切换分支
git merge sjs2 #将sjs2合并过来

git remote add origin url #绑定远程仓库
git remote -v #查看
git remote rm origin #解除绑定

git pull origin master #拉取
git push origin sjs1 #推送

git pull = git fetch + git merge
git pull --rebase = git fetch + git rebase
推荐使用git pull --rebase origin #rebase不会产生新节点 线性的 而merge是将两个分支合并到一个新的节点
